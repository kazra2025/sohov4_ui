mkdir -p ~/nodez/dl_tmp
cd ~/nodez/dl_tmp


curl --silent https://nodejs.org/dist/v24.1.0/node-v24.1.0-linux-x64.tar.gz  -o nd24.tgz
echo "4d2beaa55d962c1b31d6a4663cc831087f79f83f304a066963b65c8f4495843f  nd24.tgz" | sha256sum -c

tar -xf nd24.tgz
mv node-v24.1.0-linux-x64 nd24


mv ~/nodez/dl_tmp/nd24 ~/nodez/

# awesome, now you have nd24 in the custom location ~/nodez/


# --- set path in bashrc to get the binaries
export PATH="$HOME/nodez/nd24/bin:$PATH"

# --- now you should have
node -v        # v24.1.0
npm -v         # 11.3.0
